RequireJS test with path definitions

This example shows the use of requirejs with path definitions so the html-beautifier can be loaded with require(["html-beautify"]) 

require.config({
	paths:{
		"js-beautify": "../../lib/beautify",
		"css-beautify": "../../lib/beautify-css",
		"html-beautify": "../../lib/beautify-html"
	}
});

require(["html-beautify"],function(html_beautify){

	var input = document.getElementById("input").value;
	var output = html_beautify.html_beautify(input);

	document.getElementById("output").innerHTML = output;
});
